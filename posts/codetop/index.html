3. 无重复字符的最长子串#

滑动窗口

解题思路：

我们先从0开始取一段满足题意的子串，然后移动窗口，加入右边界触碰到重复字符则把左边界移到该重复字符上一次出现位置的右侧，并消去中途经过字符的记录，每次移动比较子串长度，最后可以得到最长子串。

func lengthOfLongestSubstring(s string) int {
    alphabet := [500]int{}
    
    // 先从0开始取一段
    l, r := 0, 0    // 在s中取[l, r)
    for ; r < len(s) && alphabet[s[r]] == 0; r++ {
        alphabet[s[r]] = r+1
    }
    // fmt.Println(s[l:r])
    res := r-l

    // 逐渐右移 l
    for l < len(s)-res {
        for ; l < alphabet[s[r]]; l++ {
            alphabet[s[l]] = 0
        }
        for ; r < len(s) && alphabet[s[r]] == 0; r++ {
            alphabet[s[r]] = r+1
        }
        // fmt.Println(s[l:r])
        if r-l > res {
            res = r-l
        }
    }

    return res
}

优化：

  1. alphabet数组大小是我自己随便定的，可以用map来降低内存消耗
  2. 代码上有冗余部分，可以优化
func lengthOfLongestSubstring(s string) int {
    alphabet := map[byte]int{}
    l, r := 0, 0    // 在s中取[l, r)
    res := 0

        // 右移l，第一次会不变

            delete(alphabet, s[l])
        }
        // 每次把r往右推到推不动为止

            alphabet[s[r]] = r+1
        }
        if r-l > res {
            res = r-l
        }
    }

    return res
}

现在代码看上去清爽一点，但不知道为什么执行时间和内存消耗都不如上一次。

15. 三数之和#

二分查找

解题思路：

题目给出的数据范围 3 <= nums.length <= 3000， 想想 $O(n^2lgn)$ 应该也能过，所以初步思路是双层遍历+二分查找。

func threeSum(nums []int) (res [][]int) {
    sort.Ints(nums)
    n := len(nums)
    for i := 0; i < n; i++ {
        if nums[i] > 0 {
            return 
        }
        if i > 0 && nums[i] == nums[i-1] {
            continue
        }
        for j := i+1; j < n; j++ {
            if j > i+1 && nums[j] == nums[j-1] {
                continue
            }
            pos := sort.SearchInts(nums[j+1:], -nums[i]-nums[j])
            if pos+j+1 < n && nums[pos+j+1] == -nums[i]-nums[j] {
                res = append(res, []int{nums[i], nums[j], nums[pos+j+1]})
            }   
        }
    }
    return res
}

结果可以说只能勉强过。

优化：

  1. 使用双指针，将时间复杂度降到 $O(n^2)$。

23. 合并K个升序链表#

链表

解题思路：

循环合并有序链表

func mergeKLists(lists []*ListNode) *ListNode {
    n := len(lists)
    if n < 1 {
        return nil
    }
    res := lists[0]
    for i := 1; i < n; i++ {
        if lists[i] != nil {
            if res == nil {
                res = lists[i]
            } else {
                res = mergeList(res, lists[i])
            }
        }
    }
    return res
}

func mergeList(list1, list2 *ListNode) *ListNode {
    if list1.Val > list2.Val {
        list1, list2 = list2, list1
    }
    head := list1
    for list2 != nil {
        for list1.Next != nil && list1.Next.Val < list2.Val {
            list1 = list1.Next
        }
        // 把 list2 对应的节点插在 list1 后面
        t := list2.Next
        list2.Next = list1.Next
        list1.Next = list2
        list2 = t
    }
    return head
}

优化思路：

分而治之

利用分治的想法，可以防止一条链表变得特别臃肿，也可以减少时间复杂度。

func mergeKLists(lists []*ListNode) *ListNode {
    n := len(lists)
    return quickMerge(lists, 0, n-1)
}

func quickMerge(lists []*ListNode, l, r int) *ListNode {
    if l > r {
        return nil
    }
    if l == r {
        return lists[l]
    }
    mid := (l+r)/2
    return mergeList(quickMerge(lists, l, mid), quickMerge(lists, mid+1, r))
}

func mergeList(list1, list2 *ListNode) *ListNode {
    if list1 == nil && list2 == nil {
        return nil
    } else if list1 == nil {
        return list2
    } else if list2 == nil {
        return list1
    }
    if list1.Val > list2.Val {
        list1, list2 = list2, list1
    }
    head := list1
    for list2 != nil {
        for list1.Next != nil && list1.Next.Val < list2.Val {
            list1 = list1.Next
        }
        // 把 list2 对应的节点插在 list1 后面
        t := list2.Next
        list2.Next = list1.Next
        list1.Next = list2
        list2 = t
    }
    return head
}

25. K 个一组翻转链表#

链表

解题思路：

利用栈的思想，把经历过的节点存储到数组中，当长度达到 k 时，全部翻转。

func reverseKGroup(head *ListNode, k int) *ListNode {
    nodeList := make([]*ListNode, 0)
    pre := &ListNode{Next:head,}
    res := pre
    for head != nil {
        nodeList = append(nodeList, head)
        head = head.Next
        if len(nodeList) == k {
            tail := nodeList[k-1].Next
            for i := k-1; i > 0; i-- {
                nodeList[i].Next = nodeList[i-1]
            }
            nodeList[0].Next = tail
            pre.Next = nodeList[k-1]
            pre = nodeList[0]
            nodeList = make([]*ListNode, 0)
        }
    }   
    return res.Next
}

优化：

可以把数组换成计数器存储，这里实现起来有点复杂，最好还是画画图直观一点。

func reverseKGroup(head *ListNode, k int) *ListNode {
    if k == 1 {
        return head
    }
    count := 0
    pre := &ListNode{Next:head,}
    res := pre
    for head != nil {
        count++
        head = head.Next
        if count == k {
            // 翻转中间 k 个节点
            p, n := pre.Next, pre.Next.Next
            for ; count > 1; count-- {
                t := n.Next
                n.Next = p 
                p = n
                n = t
            }
            count = 0

            // 调整 pre 的位置
            pre.Next.Next = head
            t := pre.Next
            pre.Next = p
            pre = t
        }
    }   
    return res.Next
}

33. 搜索旋转排序数组#

二分查找 分类讨论

解题思路：

我们把数组想象为左边半段有序和右边半段有序，通过二分查找的思想，可以分情况讨论：

  1. 中间的数小于 target，我们可以根据以下情况判断应该保留左边还是右边：
    1. 中间的数属于左有序半段，则 target 必定在右半段
    2. 中间的数属于右有序半段，并且最右端那个数大于 target，则 target 必定在右半段
    3. 中间的数属于右有序半段，并且最右端那个数也小于 target，则 target 必定在左半段
  2. 中间的数大于 target：
    1. 中间的数属于右有序半段，则 target 必定在左半段
    2. 中间的数属于左有序半段，并且最左端那个数小于 target，则 target 必定在左半段
    3. 中间的数属于左有序半段，并且最右端那个数也大于 target，则 target 必定在右半段
func search(nums []int, target int) int {
    n := len(nums)
    l, r := 0, n-1
    for l < r {
        mid := l + (r - l) / 2
        if nums[mid] == target {
            return mid
        } else if nums[mid] < target {
            if nums[l] <= nums[mid] || nums[r] >= target {
                l = mid+1
            } else {
                r = mid-1
            }
        } else if nums[l] > nums[mid] || nums[l] <= target {
            r = mid-1
        } else {
            l = mid+1
        }
    }
    if nums[l] == target {
        return l
    }
    return -1
}

41. 缺失的第一个正数#

巧思

解题思路：

由题意可知，超出数组长度的数和负数均为没有意义的数，我们可以应用桶排序的思维，把有意义的数放在他对应的桶中，使用两次遍历即可把缺失数找到。假设数组长度为 n，而数字 n 在题中也是有意义的数，我们可以在数组尾部增加一个无意义的数，相当于为 n 准备一个桶。

func firstMissingPositive(nums []int) int {
    nums = append(nums, -1)
    n := len(nums)
    for i := 0; i < n; i++ {
        for nums[i] > 0 && nums[i] < n && nums[nums[i]] != nums[i] {
            nums[i], nums[nums[i]] = nums[nums[i]], nums[i]
        }
    }
    for i := 1; i < n; i++ {
        if nums[i] != i {
            return i
        }
    }
    return n
}

42. 接雨水#

双指针

解题思路：

左视图中（从左往右看），想象最右边有一根无限高的柱子，我们只能看到依次递增的几根柱子，同时能够接住多少雨水，主要也是依靠这几根柱子，跟其他柱子关系不大。那么我们可以分为这几步解题：

  1. 找到这根近似无限高的柱子（全柱最好）
  2. 在从左往右遍历的时候找到那几根左视图中能够出现的柱子，当还没出现更高的柱子的时候，雨水主要是根据这几根柱子承接
  3. 从右往左同理

时间复杂度为 $O(n)$，空间复杂度为 $O(1)$

func trap(height []int) int {
    n := len(height)
    res := 0
    heightestpos := 0
    for i := 1; i < n; i++ {
        if height[i] > height[heightestpos] {
            heightestpos = i
        }
    }
    lmax := height[0]
    for i := 1; i < heightestpos; i++ {
        if lmax > height[i] {
            res += lmax - height[i]
        } else if lmax < height[i] {
            lmax = height[i]
        } 
    }
    rmax := height[n-1]
    for i := n-2; i > heightestpos; i-- {
        if rmax > height[i] {
            res += rmax - height[i]
        } else if rmax < height[i] {
            rmax = height[i]
        }
    }
    return res
}

54. 螺旋矩阵#

切片 引用传递

解题思路：

把螺旋打印用函数包装起来，每次打印最外面一层，同时使用行数和列数的最小值的二分之一向上取整来判断打印层数，另外记得注意只打印一行或者一列的情况。

func spiralOrder(matrix [][]int) []int {
    res := []int{}
    x := (min(len(matrix), len(matrix[0]))+1)/2
    for i := 0; i < x; i++ {
        res = append(res, layerOrder(matrix, i)...)
    }     
    return res
}

func min(x, y int) int {
    if x < y {
        return x 
    }
    return y
}

func layerOrder(matrix [][]int, t int) (res []int) {
    m, n := len(matrix), len(matrix[0])
    
    if t == n-t-1 && t == m-t-1 {
        // 如果是一个点，则直接返回一个点
        return []int{matrix[t][t]}
    } else if t == n-t-1 {
        // 如果是一列
        for i := t; i < m-t; i++ {
            res = append(res, matrix[i][n-t-1])
        }
        return res
    } else if t == m-t-1 {
        // 如果是一行
        return matrix[t][t:n-t]
    }
    for i := t; i < n-t-1; i++ {
        res = append(res, matrix[t][i])
    }
    for i := t; i < m-t-1; i++ {
        res = append(res, matrix[i][n-t-1])
    }
    for i := n-t-1; i > t; i-- {
        res = append(res, matrix[m-t-1][i])
    }
    for i := m-t-1; i > t; i-- {
        res = append(res, matrix[i][t])
    }
    return 
}

优化：

  1. 把切片的传递改成引用传递，可以大大降低内存消耗

func layerOrder(matrix [][]int, t int, res *[]int) {

    if t == n-t-1 && t == m-t-1 {
        // 如果是一个点，则直接返回一个点
        *res = append(*res, matrix[t][t])
        return
    } else if t == n-t-1 {
        // 如果是一列
        for i := t; i < m-t; i++ {
            *res = append(*res, matrix[i][n-t-1])
        }
        return
    } else if t == m-t-1 {
        // 如果是一行
        *res = append(*res, matrix[t][t:n-t]...)
        return
    }
    for i := t; i < n-t-1; i++ {
        *res = append(*res, matrix[t][i])
    }
    for i := t; i < m-t-1; i++ {

    }
    for i := n-t-1; i > t; i-- {
        *res = append(*res, matrix[m-t-1][i])
    }
    for i := m-t-1; i > t; i-- {
        *res = append(*res, matrix[i][t])
    }
    return 
}

146. LRU 缓存#

双向链表 哈希表

解题思路：

用双向链表按使用顺序存储节点们，刚使用的节点放在表头位置，越久没用的数据将慢慢移动到表尾，当存储到达容量上限的时候，直接删除表尾节点就能逐出最久未使用的数据。其中使用双向链表的原因在于，找到节点位置后，方便移动。再通过哈希表存储各个节点位置，方便快速找到要查的节点，进行删除移动操作。

type Node struct {
    Key int
    Val int
    Pre *Node
    Next *Node
}

type LRUCache struct {
    m map[int]*Node
    head *Node
    len int
    capacity int
}

func Constructor(capacity int) LRUCache {
    return LRUCache{
        // 初始化 map
        m: map[int]*Node{},
        // 设置容量上限
        capacity: capacity,
    }
}

func (this *LRUCache) Get(key int) int {
    if pos, ok := this.m[key]; ok {
        if pos == this.head {
            return pos.Val
        } 
        // 将节点拆出来，放到 head 位置
        pos.Pre.Next = pos.Next
        pos.Next.Pre = pos.Pre
        pos.Next = this.head
        pos.Pre = this.head.Pre
        this.head.Pre.Next = pos
        this.head.Pre = pos
        this.head = pos
        return pos.Val
    }
    return -1
}


func (this *LRUCache) Put(key int, value int)  {
    if this.head == nil {
        this.head = &Node {
            Key: key,
            Val: value,
        }
        this.head.Next = this.head
        this.head.Pre = this.head
        this.m[key] = this.head
        this.len++
        return 
    }

    if pos, ok := this.m[key]; ok {
        // 如果存在
        if this.head != pos {
            // 且节点并非在 head 位置
            // 将节点拆出来，放到 head 位置
            pos.Pre.Next = pos.Next
            pos.Next.Pre = pos.Pre
            pos.Pre = this.head.Pre
            pos.Next = this.head
            this.head.Pre.Next = pos
            this.head.Pre = pos
            this.head = pos
        } 
        pos.Val = value
    } else if this.len == this.capacity {
        // 容量到达上限的时候，先删除尾节点
        delete(this.m, this.head.Pre.Key)
        // 在 head 位置加节点
        this.head = this.head.Pre
        this.m[key] = this.head
        this.head.Key = key
        this.head.Val = value
    } else {
        // 容量没有到达上限，直接在头部增加新节点，并把长度加一
        this.len++
        newnode := &Node {
            Key: key,
            Val: value,
            Pre: this.head.Pre,
            Next: this.head,
        }
        this.head.Pre.Next = newnode
        this.head.Pre = newnode
        this.head = newnode
        this.m[key] = newnode
    }

    return 
}

优化：

  1. 使用双向链表操作中有很多重复代码，比如在头部增加节点、移动节点打表头，可以增加使用 ListManager 来维护双向链表

199. 二叉树的右视图#

二叉树

解题思路：

层序遍历每次选最右边的即可。

func rightSideView(root *TreeNode) []int {
    if root == nil {
        return []int{}
    }
    res := []int{}
    q := []*TreeNode{root}
    for len(q) != 0 {
        nq := []*TreeNode{}
        for _, node := range q {
            if node.Left != nil {
                nq = append(nq, node.Left)
            }
            if node.Right != nil {
                nq = append(nq, node.Right)
            }
        }
        res = append(res, q[len(q)-1].Val)
        q = nq
    }
    return res
}

300. 最长上升子序列#

动态规划

解题思路：

从数据范围来看，1 <= nums.length <= 2500 的话用 $O(n^2)$ 的 dp 方法就可以解决了。

func lengthOfLIS(nums []int) int {
    n := len(nums)
    dp := make([]int, n)
    res := 1
    for i := 0; i < n; i++ {
        dp[i] = 1
        for j := 0; j < i; j++ {
            if nums[i] > nums[j] {
                dp[i] = max(dp[i], dp[j]+1)
                res = max(res, dp[i])
            }
        }
    }
    return res
}

func max(x, y int) int {
    if x > y {
        return x
    }
    return y
}

看到题解才发现这个方法，维护一个数组 d，d[i] 代表长度为 i 的最长上升子序列的末尾元素的最小值，我们的目标是使 d[i] 上的数字尽可能地小。那么每次遍历的过程中每遇到一个新的数，尝试把它加入 d[i]，这时有两种情况，它是最大的，可以直接加在队尾；或者可以放入其中，代替一个比他大的数。最后结果就是这个数组能够有多长。最后时间复杂度为 $O(nlgn)$！

func lengthOfLIS(nums []int) int {
    n := len(nums)
    d := []int{nums[0]}
    for i := 1; i < n; i++ {
        if nums[i] > d[len(d)-1] {
            d = append(d, nums[i])
        } else {
            pos := sort.SearchInts(d, nums[i])
            d[pos] = nums[i]
        }
    }
    return len(d)
}

func getNodes(cur int, n int) int {
    if cur == 0 {
        return 1
    }
    res := 0
    left, right := cur, cur
    for left <= n {
        res += min(right, n) - left + 1
        left *= 10
        right = right * 10 + 9
    }

}

根据当前前缀 cur 进行分析：

  1. 如果这个数量小于数字 k，代表以这个数为前缀是不可行的，尝试前缀+1；

  2. 如果这个数字大于或者等于 k，代表这个前缀是可行的，即第 k 个数以这个数为前缀，尝试前缀*10。

一直如此循环到数够 k。

func findKthNumber(n int, k int) int {
    cur := 0
    for k > 0 {
        nodes := getNodes(cur, n)
        if nodes > k {
            cur *= 10
            k--
        } else {
            cur++
            k -= nodes
        }
    }
    return cur
}

2575. 找出字符串的可整除数组#

字符串 数学

解题思路：

主要是用到了数学中除法的计算方法，上一位没除尽的余数会乘上 10 加到下一位上。（幸好这题 word 范围给的是 0 到 9）

func divisibilityArray(word string, m int) []int {
    n := len(word)
    res := make([]int, n)
    prev := 0
    for i := 0; i < n; i++ {
        t := int(word[i] - '0') + prev * 10
        if t % m == 0 {
            res[i] = 1
        } else {
            res[i] = 0
        }
        prev = t % m 
    }
    return res
}

func simplifyPath(path string) string {
    stack := []string{}
    for _, name := range strings.Split(path, "/") {
        if name == ".." {
            if len(stack) > 0 {
                stack = stack[:len(stack)-1]
            }
        } else if name != "" && name != "." {
            stack = append(stack, name)
        }
    }
    return "/" + strings.Join(stack, "/")
}

func (this *Solution) GetRandom() int {
    head := this.h
    var res int

        }
        head = head.Next
    }

        stack1: list.New(),
        stack2: list.New(),
    }
}

func (this *CQueue) AppendTail(value int)  {
    this.stack1.PushBack(value)
}

func (this *CQueue) DeleteHead() int {
    // 如果第二个栈为空
    if this.stack2.Len() == 0 {
        for this.stack1.Len() > 0 {
          	// 注意这句代码
            this.stack2.PushBack(this.stack1.Remove(this.stack1.Back()))
        }
    }
    if this.stack2.Len() != 0 {
        e := this.stack2.Back()
        this.stack2.Remove(e)
        return e.Value.(int)
    }
    return -1
}
