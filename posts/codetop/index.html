3. 无重复字符的最长子串#

最初的想法是：我们先从0开始取一段满足题意的子串，然后移动窗口，加入右边界触碰到重复字符则把左边界移到该重复字符上一次出现位置的右侧，并消去中途经过字符的记录，每次移动比较子串长度，最后可以得到最长子串。

func lengthOfLongestSubstring(s string) int {
    alphabet := [500]int{}
    
    // 先从0开始取一段
    l, r := 0, 0    // 在s中取[l, r)
    for ; r < len(s) && alphabet[s[r]] == 0; r++ {
        alphabet[s[r]] = r+1
    }
    // fmt.Println(s[l:r])
    res := r-l

    // 逐渐右移 l
    for l < len(s)-res {
        for ; l < alphabet[s[r]]; l++ {
            alphabet[s[l]] = 0
        }
        for ; r < len(s) && alphabet[s[r]] == 0; r++ {
            alphabet[s[r]] = r+1
        }
        // fmt.Println(s[l:r])
        if r-l > res {
            res = r-l
        }
    }

    return res
}

优化：

  1. alphabet数组大小是我自己随便定的，可以用map来降低内存消耗
  2. 代码上有冗余部分，可以优化
func lengthOfLongestSubstring(s string) int {
    alphabet := map[byte]int{}
    l, r := 0, 0    // 在s中取[l, r)
    res := 0
   
    for l < len(s)-res {
        // 右移l，第一次会不变
        for ; l < alphabet[s[r]]; l++ {
            delete(alphabet, s[l])
        }
        // 每次把r往右推到推不动为止
        for ; r < len(s) && alphabet[s[r]] == 0; r++ {
            alphabet[s[r]] = r+1
        }
        if r-l > res {
            res = r-l
        }
    }

    return res
}

现在代码看上去清爽一点，但不知道为什么执行时间和内存消耗都不如上一次。
